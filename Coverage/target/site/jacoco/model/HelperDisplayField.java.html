package model;

public class HelperDisplayField {
    private int row;
    private int column;


    public HelperDisplayField(int row, int column) {
        this.row = row;
        this.column = column;
    }

    public int getRow() {
        return row;
    }

    public int getColumn() {
        return column;
    }

    public void setRow(int row) {
        this.row = row;
    }

    public void setColumn(int column) {
        this.column = column;
    }

    public String toString() {
        return "(" + row + ", " + column + ")";
    }

    /**
     * This method is used to print the board in the TUI.
     *
     * @param board          the board that is printed.
     * @param aiFieldDisplay
     */
    public static void printBoard(String[][] board, String[][] aiFieldDisplay) {
        for (int j = 0; j < 10; j++) {
            // print the top of the board
            System.out.print("  ");
            for (int i = 0; i < 10; i++) {
                System.out.print(" ___");
            }
            // print the numbers on each row
            System.out.println();
            if (j < 10) {
                System.out.print(j + " ");
            } else {
                System.out.print(j);
            }

            // print the board
            for (int i = 0; i < 10; i++) {
                if (board[i][j].equals(0)) {
                    System.out.print("|___");
                } else {
                    System.out.print("|_"+(String)(board[i][j]) + "_");
                }
            }

            // print the line in the right of the baord
            System.out.print("|\n");
        }

        // print the numbers/letters in the bottom
        System.out.print("  ");
        for (int i = 0; i < 10; i++) {
            System.out.print("  "+(char)(65+i) +" ");
        }
        System.out.println();
    }

}
