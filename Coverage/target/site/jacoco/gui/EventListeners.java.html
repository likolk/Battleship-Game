package gui;

import javax.swing.*;
import java.awt.event.ActionEvent;
import java.awt.event.ActionListener;
import java.awt.event.MouseAdapter;
import java.awt.event.MouseEvent;

import static model.Game.setPlayerShipsManually;

public class EventListeners extends JPanel implements ActionListener {

    // this class is going to be used to store all the event listeners
    // for the GUI

    JPanel panel = new JPanel();
    public EventListeners() {


    }

    public void mouseclicks() {
        // we will use an event listener to do this
        panel.addMouseListener(new MouseAdapter() {
            @Override
            public void mouseClicked(MouseEvent e) {
                // we will use the x and y coordinates of the mouse to place the ships
                int x = e.getX();
                int y = e.getY();
                // we will use the x and y coordinates of the mouse to place the ships
                setPlayerShipsManually(x, y);
            }
        });
    }

    @Override
    public void actionPerformed(ActionEvent e) {
        JButton btn = (JButton) e.getSource();
        System.out.println("clicked column " + btn.getClientProperty("column")
                + ", row " + btn.getClientProperty("row"));
    }






}
