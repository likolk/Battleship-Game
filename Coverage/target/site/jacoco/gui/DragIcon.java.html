package gui;

import model.Game;

import java.awt.*;
import java.awt.event.*;
import javax.swing.*;
import javax.swing.border.LineBorder;

import static java.awt.Color.*;
import static java.awt.Color.ORANGE;

public class DragIcon extends JPanel
{

    private static Game g = new Game();
    public DragIcon() {

        final JFrame frame = new JFrame("Battleship");
        TransferHandler iconHandler = new TransferHandler("icon");
        MouseListener dragListener = new DragMouseAdapter();


        JPanel panel22 = new JPanel();
        GridLayout layoutt = new GridLayout(10, 10, 1, 1);
        layoutt.setHgap(2); // horizontal gap between columns
        layoutt.setVgap(2); // vertical gap between rows
        panel22.setLayout(layoutt);
        panel22.setBorder(new LineBorder(GREEN, 1));
        panel22.setBackground(ORANGE);

        JTextField textField = new JTextField();
        textField.setText("Player 1");
        textField.setEditable(false);
        textField.setFont(new Font("Comfortaa", Font.BOLD, 20));
        textField.setHorizontalAlignment(JTextField.CENTER);
        textField.setBorder(new LineBorder(Color.BLACK, 4));
        textField.setBackground(Color.WHITE);
        textField.setVisible(true);
        // revalidate the text field
        textField.revalidate();
        for (int x = 0; x < 10; x++) {
            for (int y = 0; y < 10; y++) {
                Button[][] target = new Button[10][10];
                final Button button = new Button(x, y);
                target[y][x] = button;
                target[y][x].setBackground(CYAN);
                target[y][x].setForeground(CYAN);
                target[y][x].addActionListener(new ActionListener() {
                    public void actionPerformed(final ActionEvent e) {
                        final int[] a = button.getValues(); // get the values of the button
                        if (g.shoot(a[1], a[0])) { // if the player has hit a ship
                            button.setBackground(Color.RED); // set the button to red
                        } else if (button.getBackground() != Color.BLACK) { // if the player has not hit a ship
                            button.setBackground(Color.YELLOW); // set the button to yellow
                        }
                    }
                });
                panel22.add(target[y][x]); // add the button to the panel
            }
        }

        panel22.setBackground(CYAN);
        panel22.setBorder(new LineBorder(ORANGE, 1));
        panel22.setBackground(CYAN);
        panel22.setLayout(layoutt);


        JLabel dragLabel = new JLabel();
        dragLabel.setTransferHandler(iconHandler);
        dragLabel.addMouseListener(dragListener);
        dragLabel.setIcon(new ImageIcon("/Users/cuenc/Documents/PF2-Project/src/gui/bs.png"));
        dragLabel.setHorizontalAlignment(JLabel.CENTER);
        add(dragLabel, BorderLayout.NORTH);


        frame.add(textField, BorderLayout.NORTH);
        frame.add(panel22, BorderLayout.CENTER);
        frame.add(dragLabel, BorderLayout.SOUTH);
        frame.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
        frame.add(panel22);
        frame.pack();
        frame.setLocationRelativeTo(null);
        frame.setVisible(true);

    }



    private class DragMouseAdapter extends MouseAdapter
    {
        public void mousePressed(MouseEvent e)
        {
            JComponent c = (JComponent)e.getSource();
            TransferHandler handler = c.getTransferHandler();
            handler.exportAsDrag(c, e, TransferHandler.COPY);
        }
    }

    private static void createAndShowGUI()
    {
        JFrame frame = new JFrame("Drag Icon");
        frame.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
        frame.add(new DragIcon());
        frame.setLocationByPlatform( true );
        frame.pack();
        frame.setVisible( true );
    }

    public static void main(String[] args)
    {
        EventQueue.invokeLater(new Runnable()
        {
            public void run()
            {
                createAndShowGUI();
            }
        });
    }
}
