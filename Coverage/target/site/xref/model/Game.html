594     /**
595      * This method asks the player to make a move.
596      * @param scannerino the scanner object that is used to get the user input.
597      */
598     public static void playerTurnToMakeAMove(Scanner scannerino) {
599         int x1; // the x coordinate
600         int y1; // the y coordinate
601         boolean isValid = false; // if the move is valid
602         try { // try to get the coordinates
603             while (isValid == false) { // while the move is not valid
604                 x1 = player.shootPlayerRow(scannerino); // get the x coordinate
605                 y1 = player.shootPlayerCol(scannerino); // get the y coordinate
606                 if (aiFieldForAi.isValidMoveV2(x1, y1)) { // If the move of the ai is valid.
607                     if (aiFieldForAi.checkWhetherHitWasSuccessful(x1, y1)) { // If the move of the ai is a hit.
608                         playerScore++; // Increase the player's score.
609                         System.out.println("Ship Hit! Congratulations!");
610                         System.out.println("Current Score: " + playerScore);
611                         aiRoomsLeft--; // Decrease the number of rooms left for the ai.
612                         aiFieldForPlayer.setHit(x1, y1);  // Set the hit on the player's field.
613                         aiFieldForAi.setHit(x1, y1); // Set the hit on the ai's field.
614                     } else { // If the move of the ai is a miss.
615                         System.out.println("Ship Missed! Try again!");
616                         aiFieldForPlayer.setMissed(x1, y1); // Set the missed on the player's field.
617                         aiFieldForAi.setMissed(x1, y1); // Set the missed on the ai's field.
618                     }
619                     isValid = true; // The move is valid.
620                 }
