3   import java.awt.*;

5   public class SimpleMenuItemPainter implements MenuItemPainter {
6   
7       public Dimension getPreferredSize(Graphics2D g2d, String text) {
8           return g2d.getFontMetrics().getStringBounds(text, g2d).getBounds().getSize();
9       }
10  
11      @Override
12      public void paint(Graphics2D g2d, String text, Rectangle bounds, boolean isSelected, boolean isFocused) {
13          FontMetrics fm = g2d.getFontMetrics();
14          if (isSelected) {
15              paintBackground(g2d, bounds, Color.BLUE, Color.WHITE);
16          } else if (isFocused) {
17              paintBackground(g2d, bounds, Color.MAGENTA, Color.BLACK);
18          } else {
19              paintBackground(g2d, bounds, Color.DARK_GRAY, Color.LIGHT_GRAY);
20          }
21          int x = bounds.x + ((bounds.width - fm.stringWidth(text)) / 2);
22          int y = bounds.y + ((bounds.height - fm.getHeight()) / 2) + fm.getAscent();
23          g2d.setColor(isSelected ? Color.WHITE : Color.LIGHT_GRAY);
24          g2d.drawString(text, x, y);
25      }
26  
27      protected void paintBackground(Graphics2D g2d, Rectangle bounds, Color background, Color foreground) {
28          g2d.setColor(background);
29          g2d.fill(bounds);
30          g2d.setColor(foreground);
31          g2d.draw(bounds);
32      }
33  
34  }
