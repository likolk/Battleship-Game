735     /**
736      * This method will check whether the ship is sunk or not.
737      *
738      * @param x4 the x coordinate of the ship
739      * @param y4 the y coordinate of the ship
740      * @return true if the ship is sunk, false if not
741      */
742     public static boolean checkWhetherShipIsSunk(int x4, int y4) {
743         int row;
744         int column;
745         int count = 0;
746 
747         if (ships[x4][y4].getTypeOfShip() == "Carrier") {
748             return true;
749         } else {
750             for (int i = 0; i < ships[x4][y4].getEachShipCoordinates().size(); i += 2) {
751                 row = ships[x4][y4].getEachShipCoordinates().get(i);
752                 column = ships[x4][4].getEachShipCoordinates().get(i + 1);
753 
754                 if (!cellButtons[row][column].isEnabled()) {
755                     count++;
756                 }
757             }
758             switch (ships[x4][y4].getTypeOfShip()) {
759                 case "Carrier":
760                     if (count == 5) {
761                         return true;
762                     } else {
763                         return false;
764                     }
765                 case "Battleship":
766                     if (count == 4) {
767                         return true;
768                     } else {
769                         return false;
770                     }
771                 case "Cruiser":
772                 case "Submarine":
773                     if (count == 3) {
774                         return true;
775                     } else {
776                         return false;
777                     }
778                 case "Destroyer":
779                     if (count == 2) {
780                         return true;
781                     } else {
782                         return false;
783                     }
784                 default:
785                     return false;
786             }
787         }
788     }

807 
808             } else if (ships[xx][yy].getTypeOfShip() == "Cruiser") {

814             } else if (ships[xx][yy].getTypeOfShip() == "Destroyer") {
