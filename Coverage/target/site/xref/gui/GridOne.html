39      public class TestPane extends JPanel {
40  
41          public TestPane() {
42              setLayout(new GridLayout(11, 11));
43  
44              for (int row = 0; row < 11; row++) {
45                  if (row > 0) {
46                      add(new JLabel(Character.toString((char) (('A' + row) - 1))));
47                  } else {
48                      add(new JLabel(""));
49                  }
50                  for (int col = 0; col < 10; col++) {
51                      if (row == 0) {
52                          add(new JLabel(Integer.toString(col + 1)));
53                      } else {
54                          add(new JButton(""));
55                      }
56                  }
57              }
58          }
59          // set the background color of the panel
60          @Override
61          public void paintComponent(Graphics g) {
62              super.paintComponent(g);
63              setBackground(Color.GREEN);
64          }
